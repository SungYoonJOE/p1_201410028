def charCount(word):
    word='sangmyung university'
    d=dict()
    for c in word: 
        if c not in d:
            d[c]=1
        else:
            d[c]=d[c]+1
    print d

def countDigitsLetters(word):
    d=dict()
    d['alpha']=0
    d['digit']=0
    for c in word:
        if c.isalpha()==True:
            d['alpha']+=1
        elif c.isdigit()==True:
            d['digit']+=1
    print d 
    %matplotlib inline

    import matplotlib
    import matplotlib.pyplot as plt

    plt.bar(range(len(d)),d.values(),align='center')
    plt.xticks(range(len(d)),list(d.keys()))
    plt.show()
